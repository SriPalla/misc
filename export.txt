version: "3.8"  # Adjust the version if needed based on your Docker setup

services:
  minio:
    image: quay.io/minio/minio  # Official MinIO image
    command: server /data --console-address ":9001"
    environment:
      - MINIO_ROOT_USER=your_access_key  # Replace with your desired access key
      - MINIO_ROOT_PASSWORD=your_secret_key  # Replace with your desired secret key
    volumes:
      - minio-data:/data  # Persistent storage for MinIO data
    ports:
      - "9000:9000"  # Expose MinIO API on port 9000 (default)
      - "9001:9001"  # Optional: Expose MinIO console on port 9001

volumes:
  minio-data:  # Persistent storage volume

import java.time.LocalDateTime
import java.time.format.DateTimeFormatter

// Get current date and time
def currentDateTime = LocalDateTime.now()

// Decrement 50 minutes from current date and time
def modifiedDateTime = currentDateTime.minusMinutes(50)

// Define date format
def formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss")

// Format modified date and time as per your requirement
def modifiedDateTimeString = modifiedDateTime.format(formatter)

// Log the modified date and time
log.info("Modified Date and Time: " + modifiedDateTimeString)

// Store the modified date and time in a JMeter variable if needed
vars.put("modifiedDateTime", modifiedDateTimeString)

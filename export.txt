import org.apache.http.HttpResponse;
import org.apache.http.client.HttpClient;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

try {
    // Create HttpClient instance
    HttpClient httpClient = HttpClients.createDefault();
    
    // Define the URL of the REST endpoint
    String apiUrl = "https://api.example.com/resource";
    
    // Create HTTP GET request
    HttpGet httpGet = new HttpGet(apiUrl);

    // Execute HTTP request
    HttpResponse response = httpClient.execute(httpGet);

    // Process response
    int statusCode = response.getStatusLine().getStatusCode();
    log.info("Response code: " + statusCode);
    
    // If response is successful (status code 200)
    if (statusCode == 200) {
        // Extract response body and do further processing if needed
        String responseBody = EntityUtils.toString(response.getEntity());
        log.info("Response body: " + responseBody);
    } else {
        // Handle error response
        log.error("Error response: " + EntityUtils.toString(response.getEntity()));
    }
} catch (Exception e) {
    log.error("Exception occurred: " + e.getMessage());
}

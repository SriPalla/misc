import com.google.api.client.http.HttpTransport;
import com.google.auth.oauth2.GoogleCredentials;
import com.google.auth.oauth2.ServiceAccountCredentials;
import java.io.IOException;

class NoCredentials : GoogleCredentials() {

    @Throws(IOException::class)
    override fun getRequestMetadata() = null

    override fun hasRequestMetadata() = false

    override fun hasRequestMetadataOnly() = false

    override fun createScopedRequired() = false

    override fun createScoped(requiredScopes: MutableSet<String>?) = this

    override fun getDefaultTransport(): HttpTransport = super.getDefaultTransport()

    companion object {
        @JvmStatic
        fun create(): GoogleCredentials = NoCredentials()
    }
}

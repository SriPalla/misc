// Import necessary classes
import groovyx.net.http.RESTClient
import groovy.json.JsonSlurper

// Define the URL of the endpoint
def url = 'https://example.com/api/endpoint'

// Define the form data parameters
def formData = [
    param1: 'value1',
    param2: 'value2'
]

// Create an instance of RESTClient
def restClient = new RESTClient(url)

// Send the POST request with form data
def response = restClient.post(body: formData)

// Parse the JSON response
def jsonSlurper = new JsonSlurper()
def jsonResponse = jsonSlurper.parseText(response.data)

// Access data from the JSON response
def responseData = jsonResponse.data // Replace 'data' with the actual key you want to access

// Log or use the response data as needed
log.info("Response Data: $responseData")
